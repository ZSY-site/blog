const mongoose = require('mongoose')

const DB_NAME = 'demo'
const PORT = 27017
const IP = 'localhost'

mongoose.connect(`mongodb://${IP}:${PORT}/${DB_NAME}`, {
    useNewUrlParser: true,

    # Restful API

    一种新的api设计方法

    第一：
传统api设计 ：把每个url当做一个功能
Restful API设计: 把每个url当作一个唯一的资源


第二： 不使用url参数
传统API设计 ：/api/list?pageIndex = 2
Restful API设计 ：/api/list/2

第三： 
传统api： 用method表示操作类型
post请求 ：/api/create-blog
post请求 ：/api/update-blog?id=100
get请求 ：/api/get-blog?id =100

Restful API ： 用method表示操作类型
post请求 ：/api/blog
patch请求 （更新） ：/api/blog/100
get请求 ：/api/blog/100

const mongoose = require('../db/db')
let Schemal = mongoose.Schema

const userRules = new Schemal({
    username: {
        type: String,
        require: true,
        unique: true
    },
    password: {
        type:String,
        require:true,

    timestamps: true
})
