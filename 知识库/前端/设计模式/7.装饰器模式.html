    # 包装器

    装饰器模式是将一个对象嵌入另一个对象之中，实际上相当于这个对象被另一个对象包装起来，形成一条包装链。请求随着这条链条依次传递到所有的对象，每个对象有处理这个请求的机会。

    // 这是美式咖啡，啥都没加，只有水和咖啡，很苦很难喝
class Coffee {
    make(water) { // 黑咖啡只有水和咖啡
        return `${water}+咖啡`;
    }
    cost() { // 需要10元
        return 10;
    }
}

// 这是加奶的咖啡
class MilkCoffee {
    constructor(parent) {
        this.parent = parent;
    }
    make(water) {
        // this.parent.make(water)是原来的逻辑，原来的逻辑不变，只是增强了
        return `${this.parent.make(water)}+牛奶`;
    }
    cost() {
        return this.parent.cost() + 1;
    }
}
// 这是加糖的咖啡
class SugerCoffee {
    constructor(parent) {
        this.parent = parent;
    }
    make(water) {
        return `${this.parent.make(water)}+糖`;
    }
    cost() {
        return this.parent.cost() + 2;
    }
}
// 先拿到美式咖啡
let coffee = new Coffee();

let milkCoffee = new MilkCoffee(coffee);
let ret1 = milkCoffee.make('水')
console.log(ret1) // 输出：'水+咖啡+牛奶'

let milksugerCoffee = new SugerCoffee(milkCoffee);
let ret2 = milksugerCoffee.make('水')
console.log(ret2) // 输出： '水+咖啡+牛奶+糖'

let sugerCoffee = new SugerCoffee(coffee)
let ret3 = sugerCoffee.make('水')
console.log(ret3) // 输出：'水+咖啡+糖'
console.log(sugerCoffee.cost()) // 输出：12

    # AOP

    • 在软件业，AOP为Aspect Oriented Programming的缩写，意为：面向切面编程
    • 可以通过预编译方式和运行期动态代理实现在不修改源代码的情况下给程序动态统一添加功能的一种技术
    • AOP本质就是在函数执行之前或之后添加一些额外的逻辑，而不需要修改函数的功能

    // 买盐之前的逻辑
Function.prototype.before = function (beforeFn) {
    let _this = this;
    return function () {
        beforeFn.apply(this, arguments);
        _this.apply(this, arguments);
    }
}
Function.prototype.after = function (afterFn) {
    let _this = this;
    return function () {
        _this.apply(this, arguments);
        afterFn.apply(this, arguments);
    }
}
// 比如自己是男人，那么结婚之前，直接买
function buy(money, goods) {
    console.log(`花${money}买${goods}`);
    return 'hello'
}

// 结婚之后，你想买盐，得先想媳妇申请钱
buy = buy.before(function () {
    console.log(`向媳妇申请1块钱.`);
});

// 买完盐之后，再做剩下的后续逻辑
buy = buy.after(function () {
    console.log(`把剩下的2毛钱还给媳妇.`);
});
buy(.8, '盐');

    # 埋点

    埋点分析，是网站分析的一种常用的数据采集方法

    • 埋点方式
      • 服务器层面的：主要是通过客户端的请求进行分析
      • 客户端层面的：通过埋点进行相应的分析,方式有如下：
        • 代码埋点
        • 自动化埋点：通过AOP思想对相应的方法进行统计
        • 第三方实现 百度、友盟等...

    前端代码：

    • 当你在前端点击按钮时，后端会记录下点击了多少次
    <body>
    <button data-name="西瓜" id="watermelon">西瓜</button>
    <button data-name="苹果" id="apple">苹果</button>
    <script>
        let watermelon = document.getElementById('watermelon');
        let apple = document.getElementById('apple');

        Function.prototype.after = function (afterFn) {
            let _this = this;
            return function () {
                _this.apply(this, arguments);
                afterFn.apply(this, arguments);
            }
        }

        function click() {
            console.log('点击了' + this.dataset.name);
        }

        click = click.after(function () {
            let img = new Image();
            img.src = `http://localhost:3000/report?name=${this.dataset.name}`;
        });

        // 给每个按钮增加点击事件
        Array.from(document.querySelectorAll('button')).forEach(function (button) {
            button.addEventListener('click', click);
        });
    </script>
</body>

    # 表单校验

    • axios的interceptor就是这种
    <body>
    用户名<input type="text" name="username" id="username">
    密码<input type="text" name="password" id="password">
    <button id="submit-btn">提交</button>
    <script>
        Function.prototype.before = function (beforeFn) {
            let _this = this;
            return function () {
                let ret = beforeFn.apply(this, arguments);
                if (ret)
                    _this.apply(this, arguments);
            }
        }

        function submit() {
            alert('提交表单');
        }
        let usernameMoreThanSix = submit.before(function () {
            let username = document.getElementById('username').value;
            if (username.length < 6) {
                return alert('用户名不能少于6位');
            }
            return true;
        });
        let usernameNotNull = usernameMoreThanSix.before(function () {
            let username = document.getElementById('username').value;
            if (!username) {
                return alert('用户名不能为空');
            }
            return true;
        });

        document.getElementById('submit-btn').addEventListener('click', usernameNotNull);
    </script>
</body>
